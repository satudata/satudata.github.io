m batu = 100 gram

Vair = 150 ml

Massa jenis = m batu / Vbatu

                   = 100 g / 50 [tex]cm^{3}[/tex]

                    = 2 g/[tex]cm^{3}[/tex] atau 2000 kg/[tex]m^{3}[/tex]
